Create USERS in ORACLE

CREATE USER "STUDY1" IDENTIFIED BY "123456"
DEFAULT TABLESPACE "USERS"
TEMPORARY TABLESPACE "TEMP";
GRANT CREATE SESSION TO "STUDY1" CONTAINER = CURRENT;
GRANT CREATE TABLE TO STUDY1;
GRANT UNLIMITED TABLESPACE TO STUDY1;
grant all privileges to STUDY1 identified by 123456;
ALTER USER STUDY1 quota unlimited on USERS;
